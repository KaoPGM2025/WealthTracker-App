ก๊อปปี้รหัส code.gs ทั้งหมดด้านล่างนี้ ไปใส่ใน Google Apps Script ในเครือข่ายบัญชีชีตของคุณเพื่อเปิดระบบ Database สมบูรณ์แบบ

task_alt ขั้นตอนการติดตั้งและเชื่อมโยงระบบหลังบ้าน

  1. สร้างไฟล์ Google Sheets เปล่าใหม่ขึ้นมา 1 ไฟล์
  2. ไปที่เมนู ส่วนขยาย (Extensions) > Apps Script
  3. ลบคำสั่งทดสอบเดิมที่มีออกทั้งหมด แล้วนำคำสั่งในช่องด้านล่างนี้ไปวางแทนที่
  4. นำ ID ของ Google Sheet ของคุณ (เอาจากลิงก์ URL ค้นหาเว็บชีตของคุณ) มาแก้ไขตรงจุด YOUR_SPREADSHEET_ID_HERE ในโค้ด
  5. กดบันทึกแล้วกดที่ปุ่ม การทำให้ใช้งานได้ (Deploy) > การทำให้ใช้งานได้ใหม่ (New deployment)
  6. เลือกประเภทเป็น เว็บแอป (Web app), ตั้งค่าสิทธิ์ผู้เข้าใช้เป็น ทุกคน (Anyone) และกด Deploy
  7. คัดลอก URL ของเว็บแอป (Web app URL) ที่ได้มา ไปกรอกในหน้า การตั้งค่า (Settings) ของระบบนี้เป็นอันเสร็จสมบูรณ์!

function doPost(e) {
  var result = {};
  
  // จัดการปัญหาเรื่อง CORS และการเชื่อมต่อ
  try {
    var params = JSON.parse(e.postData.contents);
    var action = params.action;
    
    // ตั้งค่า Sheet ID ของคุณด้านล่างนี้
    var SHEET_ID = "YOUR_SPREADSHEET_ID_HERE";
    var ss = (SHEET_ID === "YOUR_SPREADSHEET_ID_HERE" || !SHEET_ID) 
             ? SpreadsheetApp.getActiveSpreadsheet() 
             : SpreadsheetApp.openById(SHEET_ID);
             
    var userSheet = getOrCreateSheet(ss, "USER");
    var savingsSheet = getOrCreateSheet(ss, "SAVINGS");
    var goalsSheet = getOrCreateSheet(ss, "GOALS");
    var settingsSheet = getOrCreateSheet(ss, "SETTINGS");

    // ================= แยกระบบตาม Actions =================
    
    // Action 1: ลงทะเบียนสมาชิก (Register)
    if (action === "register") {
      var username = params.username.toLowerCase().trim();
      var password = params.password;
      var fullName = params.fullName;
      
      // ตรวจสอบความซ้ำซ้อนของ Username
      var data = userSheet.getDataRange().getValues();
      var userExists = false;
      for(var i = 1; i < data.length; i++) {
        if(data[i][0] == username) {
          userExists = true;
          break;
        }
      }
      
      if(userExists) {
        result = { success: false, message: "ชื่อผู้ใช้นี้ถูกใช้งานแล้วในระบบ" };
      } else {
        userSheet.appendRow([username, password, fullName, new Date().toISOString()]);
        result = { success: true, message: "สมัครสมาชิกสำเร็จ" };
      }
    }
    
    // Action 2: เข้าสู่ระบบ (Login)
    else if (action === "login") {
      var username = params.username.toLowerCase().trim();
      var password = params.password;
      
      var data = userSheet.getDataRange().getValues();
      var userFound = null;
      
      for(var i = 1; i < data.length; i++) {
        if(data[i][0] == username && data[i][1] == password) {
          userFound = {
            username: data[i][0],
            fullName: data[i][2],
            joinedDate: data[i][3]
          };
          break;
        }
      }
      
      if(userFound) {
        result = { success: true, user: userFound };
      } else {
        result = { success: false, message: "ชื่อผู้ใช้หรือรหัสผ่านไม่ถูกต้อง" };
      }
    }
    
    // Action 3: ดึงข้อมูลการออมทั้งหมด (Fetch User Data)
    else if (action === "get_data") {
      var username = params.username.toLowerCase().trim();
      
      // ดึงรายการออมทั้งหมด
      var rawSavings = savingsSheet.getDataRange().getValues();
      var userSavings = [];
      for(var i = 1; i < rawSavings.length; i++) {
        if(rawSavings[i][1] == username) {
          userSavings.push({
            id: rawSavings[i][0],
            amount: parseFloat(rawSavings[i][2]),
            date: rawSavings[i][3],
            category: rawSavings[i][4],
            note: rawSavings[i][5]
          });
        }
      }
      
      // ดึงเป้าหมายออมทั้งหมด
      var rawGoals = goalsSheet.getDataRange().getValues();
      var userGoals = [];
      for(var i = 1; i < rawGoals.length; i++) {
        if(rawGoals[i][1] == username) {
          userGoals.push({
            id: rawGoals[i][0],
            name: rawGoals[i][2],
            targetAmount: parseFloat(rawGoals[i][3]),
            currentAmount: parseFloat(rawGoals[i][4]),
            deadline: rawGoals[i][5],
            category: rawGoals[i][6],
            status: rawGoals[i][7]
          });
        }
      }
      
      result = { success: true, savings: userSavings, goals: userGoals };
    }
    
    // Action 4: เพิ่มรายการออมใหม่
    else if (action === "add_saving") {
      var id = "SAVE_" + new Date().getTime();
      var username = params.username.toLowerCase().trim();
      var amount = parseFloat(params.amount);
      var date = params.date;
      var category = params.category;
      var note = params.note || "";
      
      savingsSheet.appendRow([id, username, amount, date, category, note]);
      result = { success: true, message: "บันทึกรายการออมใหม่แล้ว" };
    }
    
    // Action 5: ลบรายการออม
    else if (action === "delete_saving") {
      var id = params.id;
      var data = savingsSheet.getDataRange().getValues();
      var deleted = false;
      
      for(var i = 1; i < data.length; i++) {
        if(data[i][0] == id) {
          savingsSheet.deleteRow(i + 1);
          deleted = true;
          break;
        }
      }
      
      result = deleted ? { success: true } : { success: false, message: "ไม่พบรายการที่ต้องการลบ" };
    }
    
    // Action 6: เพิ่มเป้าหมายใหม่
    else if (action === "add_goal") {
      var id = "GOAL_" + new Date().getTime();
      var username = params.username.toLowerCase().trim();
      var name = params.name;
      var targetAmount = parseFloat(params.targetAmount);
      var currentAmount = parseFloat(params.currentAmount || 0);
      var deadline = params.deadline;
      var category = params.category;
      
      goalsSheet.appendRow([id, username, name, targetAmount, currentAmount, deadline, category, "active"]);
      result = { success: true, message: "สร้างเป้าหมายสำเร็จ" };
    }
    
    // Action 7: แก้ไข/โอนเงินออมเข้าสู่เป้าหมาย
    else if (action === "inject_goal_progress") {
      var id = params.id;
      var injectAmount = parseFloat(params.amount);
      var data = goalsSheet.getDataRange().getValues();
      var updated = false;
      
      for(var i = 1; i < data.length; i++) {
        if(data[i][0] == id) {
          var currentVal = parseFloat(data[i][4] || 0);
          var newVal = currentVal + injectAmount;
          goalsSheet.getRange(i + 1, 5).setValue(newVal);
          updated = true;
          break;
        }
      }
      
      result = updated ? { success: true } : { success: false, message: "ไม่พบเป้าหมายดังกล่าว" };
    }
    
    // Action 8: ลบเป้าหมายออมเงิน
    else if (action === "delete_goal") {
      var id = params.id;
      var data = goalsSheet.getDataRange().getValues();
      var deleted = false;
      
      for(var i = 1; i < data.length; i++) {
        if(data[i][0] == id) {
          goalsSheet.deleteRow(i + 1);
          deleted = true;
          break;
        }
      }
      result = deleted ? { success: true } : { success: false, message: "ไม่พบเป้าหมายที่ต้องการลบ" };
    }
    
  } catch(error) {
    result = { success: false, message: "Error: " + error.toString() };
  }
  
  // คืนค่าผลลัพธ์เป็น JSON กลับไปยัง Frontend
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

// ฟังก์ชันสร้างชีตเมื่อไม่มีตารางใน Google Sheets
function getOrCreateSheet(ss, sheetName) {
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    if (sheetName === "USER") {
      sheet.appendRow(["Username", "Password", "FullName", "JoinedDate"]);
    } else if (sheetName === "SAVINGS") {
      sheet.appendRow(["ID", "Username", "Amount", "Date", "Category", "Note"]);
    } else if (sheetName === "GOALS") {
      sheet.appendRow(["ID", "Username", "GoalName", "TargetAmount", "CurrentAmount", "Deadline", "Category", "Status"]);
    } else if (sheetName === "SETTINGS") {
      sheet.appendRow(["Username", "Currency", "Theme", "DailyReminder"]);
    }
  }
  return sheet;
}
